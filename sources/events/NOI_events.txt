add_namespace = noi_usa


# it's warring time
country_event = {
   id = noi_usa.1
   title = noi_usa.1.t
   desc = noi_usa.1.d
   picture = GFX_report_event_usa_heavy_cruiser

   trigger = {
        tag = USA
        has_idea = USA_enforced_neutrality
        OR = {
            has_war = yes
            date > 1941.6.1
        }
   }

   fire_only_once = yes

   option = {
        name = noi_usa.1.a
        remove_ideas = USA_enforced_neutrality
   }
}

# US annexes the Philippines
country_event = {
     id = noi_usa.2
     title = noi_usa.2.t
     desc = noi_usa.2.d
     picture = GFX_report_event_usa_heavy_cruiser

     trigger = {
         tag = USA
         has_war = yes
     }

     fire_only_once = yes

     option = {
          name = noi_usa.2.a
          annex_country = {
                target = PHI
                transfer_troops = yes
          }
     }
}

# Japan attack on Eurobros, timer for the Pacific war starts
country_event = {
    id = noi_usa.3
    title = noi_usa.3.t
    desc = noi_usa.3.d
    picture = GFX_report_event_japanese_transport_soldiers

    trigger = {
        tag = USA
        JAP = {
            has_war_with = ENG
        }
        has_war = no
    }

    fire_only_once = yes

    option = {
        name = noi_usa.3.a
        activate_mission = noi_usa_japan_war_mission
    }
}

# Sealion's happening, US joins the allies
country_event = {
    id = noi_usa.4
    title = noi_usa.4.t
    desc = noi_usa.4.d
    picture = GFX_report_event_generic_handshake

    trigger = {
        tag = USA
        any_state = {
            is_owned_by = ENG
            is_in_home_area = yes
            NOT = { is_fully_controlled_by = ENG }
        }
    }

    fire_only_once = yes

    option = {
        name = noi_usa.4.a

        IF = {
            limit = {
                has_idea = USA_enforced_neutrality
            }
            remove_ideas = USA_enforced_neutrality
        }
        ENG = {add_to_faction = USA}
    }
}
