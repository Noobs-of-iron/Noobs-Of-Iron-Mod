add_namespace = noi_sov

# Gives USSR military access to Spain if it has volunteers, because PDX can't code
country_event = {
    id = noi_sov.1
    title = noi_sov.1.t
    desc = noi_sov.1.d
    picture = GFX_report_event_generic_handshake
    fire_only_once = yes

    trigger = {
        tag = SOV
        any_country = {
            original_tag = SPR
            has_volunteers_amount_from = {
                tag = SOV
                count > 0
            }
        }
    }

    option = {
        name = noi_sov.1.a

        every_country = {
            limit = {
                original_tag = SPR
                has_volunteers_amount_from = {
                    tag = SOV
                    count > 0
                }
            }
            give_military_access = SOV
        }
    }
}
# All cities have fallen, parent trigger
country_event = {
    id = noi_sov.2
    hidden = yes
    mean_time_to_happen = { days = 1 }
    trigger = {
    
        SOV = {
            NOT = {
                controls_state = 219 #Moscow
                controls_state = 195 #Leningrad
                controls_state = 217 #Stalingrad
            }
        }
    
    }

	fire_only_once = yes

	immediate = {
		hidden_effect = {
			SOV = {
				country_event = { id = noi_sov.3 days = 1 }
			}
			every_country = {
                limit = {
                    has_war_with = SOV
                }

                country_event = { id = noi_sov.4 days = 1 }
            }
		}
	}
    


    option = {}
}
